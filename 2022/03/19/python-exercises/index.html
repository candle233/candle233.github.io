    def homework16_1():
   from openpyxl import load_workbook,Workbook
   new_wb = Workbook()
   sh = new_wb.active
   all_count= []
   for filename in ['z1.xlsx','z2.xlsx','z3.xlsx']:
       # 实例化
       wb = load_workbook(filename)
       # 激活 worksheet
       ws = wb.active
       # 记录总记录数
       count = 0
       for row in ws.rows:
         row_values = [r.value for r in row]
         row_values.append(wb.sheetnames[0])
         sh.append(row_values)
         count += int(row_values[1])
       all_count.append([wb.sheetnames[0],count])
     
     sh2 = new_wb.create_sheet('汇总')
     for r in all_count:
         sh2.append(r)
     new_wb.save('z_all.xlsx')

  37. 将重复的进⾏删除，并剩余数据保存到新的 sheet 表中

def homework20_2():
   from openpyxl import load_workbook
   wb = load_workbook('tmp.xlsx')
   ws = wb.active
   new_ws = wb.create_sheet('duplicate')
   data = []
   for row in ws.rows:
     id = row[0].value
     if id not in data:
       data.append(id)
       new_ws.append([r.value for r in row])
   wb.save('tmp.xlsx')
